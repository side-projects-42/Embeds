// Prototypes 

function Parent(attributes){
  // assign our values here
  this.name = attributes.name;
}

// abstract out our methods
Parent.prototype.methodName = function(){
  // return something here
}

// child skeleton syntax 

function Child(attributes){
  Parent.call(this, attributes);
  // any special attributes go here
}

// tell the child to inherit the methods too

Child.prototype = Object.create(Parent.prototype);

// any special methods of the child
Child.prototype.methodName = function(){
  
}

// Class Syntax Skeleton 

class Parent{
  constructor(attributes){
    //assigning our attributes here
    this.name = attributes.name;
  }
  // methods go here
  methodName(){
    //return something here
  }
}

//child syntax skeleton 

class Child extends Parent{
  constructor(attributes){
    super(attributes);
    // any special attributes for the child go here
  }
  // any special methods for the child go here
  methodName(){
    //return something
  }
}

// Objects 
// another fundamental building block of JS
// allow us to group together properties and values 
// order of properties doesn't matter 
// Everything in JavaScript is an object! Objects are used for collections of data 

// // anatomy of an object 

const dog = {
  // key value pairs (properties:values)
  name: 'Ada',
  breed: 'bali dog',
  age: 3, 
  colour: 'brindle',
  favFood: 'Salmon',
  // methods - functions that belong to objects 
  eat: function(){
    // string template literal 
    return `${this.name} likes to eat ${this.favFood}`;
  }
}
// // notice that it does not return the key value pairs to us in the same order - order doesn't matter in an object - if you want order to matter then you want to use an array. 

console.log(dog);

// // // how do we access things within an object? for example how can I access the dogs name? 
// // // dot notation 
console.log(dog.name);

// // brack notation - this would be useful if you had keys with spaces in them
// // could get data from another language for example
console.log(dog['name']);

// // // implicit binding (we will talk about this more next week, but basically when the function is invoked we look to the left of the dot and that's what this refers to - this is about 80% of use cases and it applies to objects with methods)

console.log(dog.eat());

// // nesting objects

  // key value pairs (properties:values)
  name: 'Ada',
  breed: 'bali dog',
  age: 3, 
  colour: 'brindle',
  favFood: 'Salmon',
  sister: {
    name: 'Noa',
    breed: 'bali dog',
    age: '8 months',
},
  
  eat: function(){
    // string template literal 
    return `${this.name} likes to eat ${this.favFood}`;
  }
}


console.log(dog.sister.name);
